Når brugeren har skrevet sin alder og trykker på knappen, så skal han have et svar i #resultat, der svarer til hans alder:

  • under 6 = "På vej"
  • 6-13 = "Undervisningspligt"
  • 13-15 = "arbejde i fritiden"
  • 15-17 = "fuldtidsarbejde"
  • 17 = "Bilkørekort"
  • 18-21 = "Fyrværkeri og tobak"
  • 21-24 = "Kørekort til bus og lastbil"
  • 24-65 = "Kørekort til stor motorcykel"
  • over 65 = "Folkepension"
